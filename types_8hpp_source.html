180 [[nodiscard]] inline double distance(const Event_<T> &e, const uint8_t type = DISTANCE_NORM_L2 | DISTANCE_FLAG_SPATIAL) const {
181 if(static_cast<bool>(type & DISTANCE_FLAG_SPATIOTEMPORAL)) {
182 return cv::norm(cv::Matx<T, 3, 1>(Event_<T>::x - e.x, Event_<T>::y - e.y, Event_<T>::t - e.t), type & 0x0F);
183 }
184 if(static_cast<bool>(type & DISTANCE_FLAG_SPATIAL) || !static_cast<bool>(type & 0xF0)) {
185 return cv::norm(cv::Matx<T, 2, 1>(Event_<T>::x - e.x, Event_<T>::y - e.y), type & 0x0F);
186 }
187 if(static_cast<bool>(type & DISTANCE_FLAG_TEMPORAL)) {
188 return Event_<T>::t - e.t;
